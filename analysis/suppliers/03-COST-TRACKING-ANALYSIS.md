  startDate: Date,
  endDate: Date,
  businessId: string

    productVariantId,
    period: { startDate, endDate },
    totalRevenue,
    totalCost,
    totalProfit,
    profitMargin,
    unitsSold: transactions.reduce((sum, t) => sum + Math.abs(t.quantity), 0),
